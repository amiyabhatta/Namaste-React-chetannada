function Timer() {
  const [count, setCount] = useState(0);
  console.log("ok");
  useEffect(() => {
    console.log("effect");
    return () => {
      console.log("last from effect");
    };
  }, [count]);
  console.log("ok1");
  const handleButtonClick = () => {
    // setCount(count+1);
    setCount((k) => k + 1);
    setCount((k) => k + 1);
  };
  console.log("ok2");
  return (
    <>
      <h1>I have rendered {count} times!</h1>
      <button onClick={handleButtonClick}>Click me</button>
      {console.log("last")}
    </>
  );
}
